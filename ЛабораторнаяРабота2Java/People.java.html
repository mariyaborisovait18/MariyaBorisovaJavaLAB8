import java.util.Scanner;

public class People {

    GlobalVerification mistakePeople = new GlobalVerification();

    private String login;
    private int height;
    private static Scanner scanner = new Scanner(System.in);

    // Конструктор для реализации самого задания 1.2
    public People() {
        System.out.print("Введите имя: ");
        this.login = mistakePeople.stringWithoutNum();

        System.out.print("Введите рост: ");
        this.height = mistakePeople.validateHeight();
    }

    // Конструктор задания 2
    // Конструктор для реализации задания 2
    public People(boolean noHeight) {
        this.height = 0;
        this.login = "";
    }
    // Конструктор задания 2
    // Конструктор для реализации задания 2
    public People(int height) {
        this.height = mistakePeople.height(height);
        this.login = "";
    }

    // Свойства - без деления по конструкторам
    public int getHeight() {
        return height;
    }

    /**
     * Метод формирует текстовое представление роста объекта
     * @return Пример: рост 167
     * */
    @Override
    public String toString() {
        return " рост " + height;
    }
}
