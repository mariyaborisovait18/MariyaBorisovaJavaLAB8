public class City {
    private String name;
    private City[] connectedCities;
    private int[] costs;
    private int routeCount;

    // Конструктор задания 3
    public City(String name) {
        this.name = name;
        this.connectedCities = new City[10]; // массив на 10 городов
        this.costs = new int[10]; // массив на 10 стоимостей
        this.routeCount = 0;
    }

    // Свойства
    // Получение названия города
    public String getName() {
        return name;
    }

    // Получение количества маршрутов
    public int getRouteCount() {
        return routeCount;
    }

    // Получение города по индексу
    public City getConnectedCity(int index) {
        return connectedCities[index];
    }

    // Получение стоимости по индексу
    public int getCost(int index) {
        return costs[index];
    }

    /**
     * Метод добавляет новый маршрут из текущего города в другой город с указанной стоимостью.
     * @param city город назначения
     * @param cost стоимость проезда
     * @return Пример:Город 'A' - (F: 1)
     * */
    public void addRoute(City city, int cost) {
        if (routeCount < 10) {
            connectedCities[routeCount] = city;
            costs[routeCount] = cost;
            routeCount++;
        } else {
            System.out.println("Нельзя добавить больше маршрутов!");
        }
    }
    /**
     * Метод формирует текстовое представление города и маршрутов со стоимостью в другие города
     * @return Пример:Город 'A' - (F: 1, D: 6, B: 5)
     * */
    @Override
    public String toString() {
        String result = "Город '" + name + "'";
        if (routeCount == 0) {
            result += " - нет маршрутов в другие города";
        } else {
            result += " - (";
            for (int i = 0; i < routeCount; i++) {
                if (i > 0) {
                    result += ", ";
                }
                result += connectedCities[i].getName() + ": " + costs[i];
            }
            result += ")";
        }
        return result;
    }
}
