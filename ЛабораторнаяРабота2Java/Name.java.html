import java.util.Scanner;

public class Name {
    Scanner scanner = new Scanner(System.in);
    GlobalVerification mistakeName = new GlobalVerification();
    private String name;
    private String surname;
    private String middleName;

    // конструктор задания 1
    public Name() {
        System.out.print("Введите имя: ");
        this.name = mistakeName.stringWithoutNum();

        System.out.print("Введите фамилию: ");
        this.surname = mistakeName.stringWithoutNum();

        System.out.print("Введите отчество: ");
        this.middleName = mistakeName.stringWithoutNum();
    }

    // Конструктор задания 4.5
    // Конструктор только с личным именем
    public Name(String name) {
        this.name = name;
        this.surname = "";
        this.middleName = "";
    }

    // Конструктор задания 4.5
    // Конструктор с личным именем и фамилией
    public Name(String name, String surname) {
        this.name = name;
        this.surname = surname;
        this.middleName = "";
    }

    // Конструктор задания 4.5
    // Конструктор со всеми тремя параметрами
    public Name(String name, String surname, String middleName) {
        this.name = mistakeName.stringWithoutNum();
        this.surname = mistakeName.stringWithoutNum();
        this.middleName = middleName;
    }

    // Инкапсуляция - без деления по конструкторам
    public String getName() {
        return name;
    }
    public String getSurname() {
        return surname;
    }
    public String getMiddleName() {
        return middleName;
    }
    public void setSurname(String surname) {
        this.surname = surname;
    }
    public void setMiddleName(String middleName) {
        this.middleName = middleName;
    }

    /**
     *  Метод вывода введённых данных
     * @return Имя Фамилию и Отчество по наличию текущих
     * */
    @Override
    public String toString() {
        StringBuilder result = new StringBuilder();

        // Добавляем фамилию, если она не пустая
        if (surname != null && !surname.trim().isEmpty()) {
            result.append(surname);
        }

        // Добавляем имя, если оно не пустое
        if (name != null && !name.trim().isEmpty()) {
            if (result.length() > 0) {
                result.append(" ");
            }
            result.append(name);
        }

        // Добавляем отчество, если оно не пустое
        if (middleName != null && !middleName.trim().isEmpty()) {
            if (result.length() > 0) {
                result.append(" ");
            }
            result.append(middleName);
        }

        return result.toString();
    }
}
