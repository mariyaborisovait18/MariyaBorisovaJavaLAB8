import java.util.Scanner;
public class GlobalVerification {
    Scanner scanner = new Scanner(System.in);

    /**
     * Метод формирует подменю основного меню
     * @param number  вариант меню, который ведёт к определённому действию
     * @return Пример: 1 - вывод основного МЕНЮ
     * */
    public int meni(int number) {
        System.out.println("Выберите действие");
        System.out.println("1 - МЕНЮ");
        System.out.println("2 - повторить задание");
        System.out.println("0 - ВЫХОД");

        int choice = scanner.nextInt();
        switch (choice) {
            case 1:
                return -1;
            case 2:
                return number;
            case 0:
                return 0;
            default:
                System.out.println("Некорректный выбор. Возврат в меню.");
                return -1;
        }
    }

    /**
     * Метод проверяет корректность введённого роста
     * @param number - проверяемый рост
     * @return Пример: рост = 12 - 'Введите корректный рост человека'
     * */
    public int height(int number) {
        while (number < 140 || number > 240) {
            System.out.println("Введите корректный рост человека");
            number = scanner.nextInt();
        }
        return number;
    }

    /**
     * Метод проверяет положительность числа
     * @param number - проверяемое число
     * @return Пример: число = 0 - 'Введите положительное число от 1'
     * */
    public int numberNum(int number) {
        while (number < 1) {
            System.out.println("Введите положительное число от 1");
            number = scanner.nextInt();
        }
        return number;
    }

    /**
     * Метод проверяет наличие цифр в строке
     * @return Пример: строка  = 020917даодшыралы - 'Есть цифры! Введите строку заново: '
     * */
    public String stringWithoutNum() {
        while (true) {
            String input = scanner.nextLine();

            // Проверяем, есть ли в строке цифры
            if (!input.matches(".*\\d.*")) {
                return input; // возвращаем строку
            } else {
                System.out.println("Есть цифры! Введите строку заново: ");
            }
        }
    }

    /**
     * Метод проверяет корректность ввода роста
     * @return Пример: рост = 12 - 'Некорректный рост . . . Введите корректный рост (140-230 см):'
     * */
    public int validateHeight() {
        int height = scanner.nextInt();
        while (height < 140 || height > 230) {
            System.out.print("Некорректный рост " + height + ". Введите корректный рост (140-230 см): ");
            height = scanner.nextInt();
        }
        return height;
    }

}
