public class Cat {
    private String name;

    // Конструктор задания 5
    public Cat(String name) {
        this.name = name;
    }

    // Свойства
    public String getName() {
        return name;
    }

    /**
     * Метод формирует текстовое представление кота
     * @return Пример: кот Барсик
     * */
    @Override
    public String toString() {
        return "кот: " + name;
    }

    /**
     * Метод формирует текстовое представление одного "мяу"
     * @return Пример: Барсик: мяу!
     * */
    public void meow() {
        System.out.println(name + ": мяу!");
    }

    /**
     * Метод формирует текстовое представление n-го количества "мяу"
     * @param n количество "мяу", которые нужно вывести, если n <= 0, то выводится " . . . "
     * @return Пример: Барсик: мяу-мяу-мяу-мяу-мяу-мяу-мяу-мяу-мяу-мяу!
     * */
    public void meow(int n) {
        if (n <= 0) {
            System.out.println(name + ": ...!");
            return;
        }
        StringBuilder meowString = new StringBuilder();

        for (int i = 0; i < n; i++) {
            if (i > 0) {
                System.out.print ("-");
            }
            System.out.print ("мяу");
        }
        System.out.print("!");
        System.out.println(name + ": " + meowString.toString());
    }
}
